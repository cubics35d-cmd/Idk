    ⚙️ Plugin Loader
    
  

  

    Full plugin API — register commands, message listeners, intervals. Scripts get Plugin and Bot injected automatically.
  

  

    📖 API Reference
    
Plugin.registerCommand(name, { usage, desc, handler(m, args, ctx) })
Plugin.registerListener(async (m, ctx) => { })
Plugin.registerInterval(fn, ms)
Plugin.registerTimeout(fn, ms)

ctx.send(text, replyTo?)   — send, replies to trigger msg by default
ctx.sendRaw(text)          — send with NO reply (use for GIFs/media!)
ctx.sendFile(blob, name, replyTo?) — upload a Blob as a real Discord attachment
ctx.renderHTML(html)       — render HTML string → PNG blob (use  for size)
ctx.edit(msgId, text)      — edit a message
ctx.react(emoji)           — react to trigger msg
ctx.del(msgId?)            — delete msg

Commands trigger on BOTH /c void AND /void (bare prefix)

// ctx object inside handlers:
ctx.send(text, replyTo?)   — send message
ctx.edit(msgId, text)      — edit message
ctx.react(emoji)           — react to trigger msg
ctx.del(msgId?)            — delete msg
ctx.log(text)              — log panel
ctx.user                   — sender's eco data
ctx.token / ctx.channel

// args object (from command text after cmd name):
args[0], args[1]...        — space-split tokens
args.raw                   — full string
args.mentions              — ['userId', ...]
args.numbers               — [123, ...]

// Bot global (usable anywhere):
Bot.send(ch, text, replyTo?)
Bot.sendAs(token, ch, text)
Bot.getCredits(userId)
Bot.addCredits(userId, name, amount)
Bot.setCredits(userId, amount)
Bot.addExp(userId, name, amount)
Bot.getUser(userId, name)
Bot.getAll()               — all ecoCredits
Bot.getToken()             — active bot token
Bot.getChannel()           — current channel id
Bot.getPrefix()            — cmdPrefix
Bot.random(array)          — random array item
Bot.credits                — ecoCredits ref (live)
Bot.shop                   — ecoShop ref (live)
